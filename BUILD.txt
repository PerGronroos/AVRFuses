To build libavrdude.a we need to build Universal binaries of libusb, 
libusb-compat and libavrdude. For some reason the configure process in avrdude
doesn't find libusb when it's compiled static only, so first compile
libusb and libusb-compat with:
./configure --prefix=/Users/jason/avrdude --disable-dependency-tracking CFLAGS="-arch i386 -arch ppc -arch x86_64" LDFLAGS="-arch i386 -arch ppc -arch x86_64"
./configure --prefix=/Users/jason/avrdude --disable-dependency-tracking CFLAGS="-arch i386 -arch ppc -arch x86_64" LDFLAGS="-arch i386 -arch ppc -arch x86_64" PKG_CONFIG_PATH=/Users/jason/avrdude/lib/pkgconfig

and avrdude with:
export ARCH="ppc"
export CFLAGS="-arch $ARCH  -isysroot /Developer/SDKs/MacOSX10.5.sdk -mmacosx-version-min=10.5"
export CPPFLAGS="-I/Users/jason/avrdude/include"
export LDFLAGS="-L/Users/jason/avrdude/lib"

make clean
./configure --prefix=/Users/jason/avrdude
make
mv libavrdude.a libavrdude-$ARCH.a

export ARCH="i386"
export CFLAGS="-arch $ARCH  -isysroot /Developer/SDKs/MacOSX10.5.sdk -mmacosx-version-min=10.5"
export CPPFLAGS="-I/Users/jason/avrdude/include"
export LDFLAGS="-L/Users/jason/avrdude/lib"

make clean
./configure --prefix=/Users/jason/avrdude
make
mv libavrdude.a libavrdude-$ARCH.a

export ARCH="x86_64"
export CFLAGS="-arch $ARCH  -isysroot /Developer/SDKs/MacOSX10.5.sdk -mmacosx-version-min=10.5"
export CPPFLAGS="-I/Users/jason/avrdude/include"
export LDFLAGS="-L/Users/jason/avrdude/lib"

make clean
./configure --prefix=/Users/jason/avrdude
make
mv libavrdude.a libavrdude-$ARCH.a

lipo libavrdude-ppc.a libavrdude-x86_64.a libavrdude-i386.a -create -output libavrdude.a


and then recompile libusb and libusb-compat with:
./configure --prefix=/Users/jason/avrdude --disable-dependency-tracking CFLAGS="-arch i386 -arch ppc -arch x86_64" LDFLAGS="-arch i386 -arch ppc -arch x86_64" --disable-shared
./configure --prefix=/Users/jason/avrdude --disable-dependency-tracking CFLAGS="-arch i386 -arch ppc -arch x86_64" LDFLAGS="-arch i386 -arch ppc -arch x86_64" PKG_CONFIG_PATH=/Users/jason/avrdude/lib/pkgconfig --disable-shared
